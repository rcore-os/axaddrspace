9/// A physical frame which will be automatically deallocated when dropped.
10///
11/// The frame is allocated using the [`AxMmHal`] implementation. The size of the frame is likely to
12/// be 4 KiB but the actual size is determined by the [`AxMmHal`] implementation.
13#[derive(Debug)]
14pub struct PhysFrame<H: AxMmHal> {
15    start_paddr: Option<HostPhysAddr>,
16    _marker: PhantomData<H>,
17}
18
19impl<H: AxMmHal> PhysFrame<H> {
20    /// Allocate a [`PhysFrame`].
21    pub fn alloc() -> AxResult<Self> {
22        let start_paddr = H::alloc_frame()
23            .ok_or_else(|| ax_err_type!(NoMemory, "allocate physical frame failed"))?;
24        assert_ne!(start_paddr.as_usize(), 0);
25        Ok(Self {
26            start_paddr: Some(start_paddr),
27            _marker: PhantomData,
28        })
29    }
30
31    /// Allocate a [`PhysFrame`] and fill it with zeros.
32    pub fn alloc_zero() -> AxResult<Self> {
33        let mut f = Self::alloc()?;
34        f.fill(0);
35        Ok(f)
36    }
37
38    /// Create an uninitialized [`PhysFrame`].
39    ///
40    /// # Safety
41    ///
42    /// The caller must ensure that the [`PhysFrame`] is only used as a placeholder and never
43    /// accessed.
44    pub const unsafe fn uninit() -> Self {
45        Self {
46            start_paddr: None,
47            _marker: PhantomData,
48        }
49    }
50
51    /// Get the starting physical address of the frame.
52    pub fn start_paddr(&self) -> HostPhysAddr {
53        self.start_paddr.expect("uninitialized PhysFrame")
54    }
55
56    /// Get a mutable pointer to the frame.
57    pub fn as_mut_ptr(&self) -> *mut u8 {
58        H::phys_to_virt(self.start_paddr()).as_mut_ptr()
59    }
60
61    /// Fill the frame with a byte. Works only when the frame is 4 KiB in size.
62    pub fn fill(&mut self, byte: u8) {
63        unsafe { core::ptr::write_bytes(self.as_mut_ptr(), byte, PAGE_SIZE) }
64    }
65}
66
67impl<H: AxMmHal> Drop for PhysFrame<H> {
68    fn drop(&mut self) {
69        if let Some(start_paddr) = self.start_paddr {
70            H::dealloc_frame(start_paddr);
71            debug!("[AxVM] deallocated PhysFrame({start_paddr:#x})");
72        }
73    }
74}

84    #[test]

86    fn test_alloc_dealloc_cycle() {
87        let frame = PhysFrame::<MockHal>::alloc()
88            .unwrap_or_else(|e| panic!("Failed to allocate frame: {:?}", e));
89        assert_eq!(frame.start_paddr().as_usize(), BASE_PADDR);
90        // frame is dropped here, dealloc_frame should be called
91    }

93    #[test]
94    #[axin(decorator(mock_hal_test), on_exit(test_dealloc_count(1)))]
95    fn test_alloc_zero() {
96        let frame = PhysFrame::<MockHal>::alloc_zero()
97            .unwrap_or_else(|e| panic!("Failed to allocate zero frame: {:?}", e));
98        assert_eq!(frame.start_paddr().as_usize(), BASE_PADDR);
99        let ptr = frame.as_mut_ptr();
100        let page = unsafe { &*(ptr as *const [u8; PAGE_SIZE]) };
101        assert!(page.iter().all(|&x| x == 0));
102    }

104    #[test]
105    #[axin(decorator(mock_hal_test), on_exit(test_dealloc_count(1)))]
106    fn test_fill_operation() {
107        let mut frame = PhysFrame::<MockHal>::alloc()
108            .unwrap_or_else(|e| panic!("Failed to allocate frame: {:?}", e));
109        assert_eq!(frame.start_paddr().as_usize(), BASE_PADDR);
110        frame.fill(0xAA);
111        let ptr = frame.as_mut_ptr();
112        let page = unsafe { &*(ptr as *const [u8; PAGE_SIZE]) };
113        assert!(page.iter().all(|&x| x == 0xAA));
114    }
115
116    #[test]
117    #[axin(decorator(mock_hal_test), on_exit(test_dealloc_count(5)))]
118    fn test_fill_multiple_frames() {
119        const NUM_FRAMES: usize = 5;
120
121        let mut frames = Vec::new();
122        let mut patterns = Vec::new();
123
124        for i in 0..NUM_FRAMES {
125            let mut frame = PhysFrame::<MockHal>::alloc().unwrap();
126            let pattern = (0xA0 + i) as u8;
127            frame.fill(pattern);
128            frames.push(frame);
129            patterns.push(pattern);
130        }
131
132        for i in 0..NUM_FRAMES {
133            let actual_page = unsafe { &*(frames[i].as_mut_ptr() as *mut [u8; PAGE_SIZE]) };
134            let expected_page = &[patterns[i]; PAGE_SIZE];
135
136            assert_eq!(
137                actual_page, expected_page,
138                "Frame verification failed for frame index {i}: Expected pattern 0x{:02x}",
139                patterns[i]
140            );
141        }
142    }
