1//! Unified guest memory access interface
2//!
3//! This module provides a safe and consistent way to access guest memory
4//! from VirtIO device implementations, handling address translation and
5//! memory safety concerns.
6use crate::GuestPhysAddr;
7use axerrno::{AxError, AxResult};
8use memory_addr::PhysAddr;
9
10/// A stateful accessor to the memory space of a guest
11pub trait GuestMemoryAccessor {
12    /// Translate a guest physical address to host physical address and get access limit
13    ///
14    /// Returns a tuple of (host_physical_address, accessible_size) if the translation
15    /// is successful. The accessible_size indicates how many bytes can be safely
16    /// accessed starting from the given guest address.
17    fn translate_and_get_limit(&self, guest_addr: GuestPhysAddr) -> Option<(PhysAddr, usize)>;
18
19    /// Read a value of type V from guest memory
20    ///
21    /// # Returns
22    ///
23    /// Returns `Err(AxError::InvalidInput)` in the following cases:
24    /// - The guest address cannot be translated to a valid host address
25    /// - The accessible memory region starting from the guest address is smaller
26    ///   than the size of type V (insufficient space for the read operation)
27    ///
28    /// # Safety
29    ///
30    /// This function uses volatile memory access to ensure the read operation
31    /// is not optimized away by the compiler, which is important for device
32    /// register access and shared memory scenarios.
33    fn read_obj<V: Copy>(&self, guest_addr: GuestPhysAddr) -> AxResult<V> {
34        let (host_addr, limit) = self
35            .translate_and_get_limit(guest_addr)
36            .ok_or(AxError::InvalidInput)?;
37
38        // Check if we have enough space to read the object
39        if limit < core::mem::size_of::<V>() {
40            return Err(AxError::InvalidInput);
41        }
42
43        unsafe {
44            let ptr = host_addr.as_usize() as *const V;
45            Ok(core::ptr::read_volatile(ptr))
46        }
47    }
48
49    /// Write a value of type V to guest memory
50    ///
51    /// # Returns
52    ///
53    /// Returns `Err(AxError::InvalidInput)` in the following cases:
54    /// - The guest address cannot be translated to a valid host address
55    /// - The accessible memory region starting from the guest address is smaller
56    ///   than the size of type V (insufficient space for the write operation)
57    ///
58    /// # Safety
59    ///
60    /// This function uses volatile memory access to ensure the write operation
61    /// is not optimized away by the compiler, which is important for device
62    /// register access and shared memory scenarios.
63    fn write_obj<V: Copy>(&self, guest_addr: GuestPhysAddr, val: V) -> AxResult<()> {
64        let (host_addr, limit) = self
65            .translate_and_get_limit(guest_addr)
66            .ok_or(AxError::InvalidInput)?;
67
68        // Check if we have enough space to write the object
69        if limit < core::mem::size_of::<V>() {
70            return Err(AxError::InvalidInput);
71        }
72
73        unsafe {
74            let ptr = host_addr.as_usize() as *mut V;
75            core::ptr::write_volatile(ptr, val);
76        }
77        Ok(())
78    }
79
80    /// Read a buffer from guest memory
81    fn read_buffer(&self, guest_addr: GuestPhysAddr, buffer: &mut [u8]) -> AxResult<()> {
82        if buffer.is_empty() {
83            return Ok(());
84        }
85
86        let (host_addr, accessible_size) = self
87            .translate_and_get_limit(guest_addr)
88            .ok_or(AxError::InvalidInput)?;
89
90        // Check if we can read the entire buffer from this accessible region
91        if accessible_size >= buffer.len() {
92            // Simple case: entire buffer fits within accessible region
93            unsafe {
94                let src_ptr = host_addr.as_usize() as *const u8;
95                core::ptr::copy_nonoverlapping(src_ptr, buffer.as_mut_ptr(), buffer.len());
96            }
97            return Ok(());
98        }
99
100        // Complex case: buffer spans multiple regions, handle region by region
101        let mut current_guest_addr = guest_addr;
102        let mut remaining_buffer = buffer;
103
104        while !remaining_buffer.is_empty() {
105            let (current_host_addr, current_accessible_size) = self
106                .translate_and_get_limit(current_guest_addr)
107                .ok_or(AxError::InvalidInput)?;
108
109            let bytes_to_read = remaining_buffer.len().min(current_accessible_size);
110
111            // Read from current accessible region
112            unsafe {
113                let src_ptr = current_host_addr.as_usize() as *const u8;
114                core::ptr::copy_nonoverlapping(
115                    src_ptr,
116                    remaining_buffer.as_mut_ptr(),
117                    bytes_to_read,
118                );
119            }
120
121            // Move to next region
122            current_guest_addr =
123                GuestPhysAddr::from_usize(current_guest_addr.as_usize() + bytes_to_read);
124            remaining_buffer = &mut remaining_buffer[bytes_to_read..];
125        }
126
127        Ok(())
128    }
129
130    /// Write a buffer to guest memory
131    fn write_buffer(&self, guest_addr: GuestPhysAddr, buffer: &[u8]) -> AxResult<()> {
132        if buffer.is_empty() {
133            return Ok(());
134        }
135
136        let (host_addr, accessible_size) = self
137            .translate_and_get_limit(guest_addr)
138            .ok_or(AxError::InvalidInput)?;
139
140        // Check if we can write the entire buffer to this accessible region
141        if accessible_size >= buffer.len() {
142            // Simple case: entire buffer fits within accessible region
143            unsafe {
144                let dst_ptr = host_addr.as_usize() as *mut u8;
145                core::ptr::copy_nonoverlapping(buffer.as_ptr(), dst_ptr, buffer.len());
146            }
147            return Ok(());
148        }
149
150        // Complex case: buffer spans multiple regions, handle region by region
151        let mut current_guest_addr = guest_addr;
152        let mut remaining_buffer = buffer;
153
154        while !remaining_buffer.is_empty() {
155            let (current_host_addr, current_accessible_size) = self
156                .translate_and_get_limit(current_guest_addr)
157                .ok_or(AxError::InvalidInput)?;
158
159            let bytes_to_write = remaining_buffer.len().min(current_accessible_size);
160
161            // Write to current accessible region
162            unsafe {
163                let dst_ptr = current_host_addr.as_usize() as *mut u8;
164                core::ptr::copy_nonoverlapping(remaining_buffer.as_ptr(), dst_ptr, bytes_to_write);
165            }
166
167            // Move to next region
168            current_guest_addr =
169                GuestPhysAddr::from_usize(current_guest_addr.as_usize() + bytes_to_write);
170            remaining_buffer = &remaining_buffer[bytes_to_write..];
171        }
172
173        Ok(())
174    }
175
176    /// Read a volatile value from guest memory (for device registers)
177    fn read_volatile<V: Copy>(&self, guest_addr: GuestPhysAddr) -> AxResult<V> {
178        self.read_obj(guest_addr)
179    }
180
181    /// Write a volatile value to guest memory (for device registers)
182    fn write_volatile<V: Copy>(&self, guest_addr: GuestPhysAddr, val: V) -> AxResult<()> {
183        self.write_obj(guest_addr, val)
184    }
185}

200    impl MockTranslator {

202            Self {
203                base_addr,
204                memory_size,
205            }
206        }

209    impl GuestMemoryAccessor for MockTranslator {
210        fn translate_and_get_limit(&self, guest_addr: GuestPhysAddr) -> Option<(PhysAddr, usize)> {
211            // Simple mapping: guest address directly maps to mock memory region
212            let offset = guest_addr.as_usize();
213            if offset < self.memory_size {
214                // Convert physical address to virtual address for actual memory access
215                let phys_addr =
216                    PhysAddr::from_usize(BASE_PADDR + self.base_addr.as_usize() + offset);
217                let virt_addr = crate::test_utils::MockHal::mock_phys_to_virt(phys_addr);
218                let accessible_size = self.memory_size - offset;
219                Some((PhysAddr::from_usize(virt_addr.as_usize()), accessible_size))
220            } else {
221                None
222            }
223        }
224    }

250
251        // Test buffer read/write operations

255        // Write buffer

259

280    #[test]
281    #[axin(decorator(mock_hal_test))]
282    fn test_two_vm_isolation() {
283        // Create two different translators to simulate two different VMs
284        let vm1_translator =
285            MockTranslator::new(PhysAddr::from_usize(0), crate::test_utils::MEMORY_LEN / 2); // Offset for VM1
286        let vm2_translator = MockTranslator::new(
287            PhysAddr::from_usize(crate::test_utils::MEMORY_LEN / 2),
288            crate::test_utils::MEMORY_LEN,
289        ); // Offset for VM2
290
291        // Both VMs write to the same guest address but different host memory regions
292        let guest_addr = GuestPhysAddr::from_usize(0x100);
293        let vm1_data: u64 = 0xDEADBEEFCAFEBABE;
294        let vm2_data: u64 = 0x1234567890ABCDEF;
295
296        // VM1 writes its data
297        vm1_translator
298            .write_obj(guest_addr, vm1_data)
299            .expect("VM1 failed to write data");
300
301        // VM2 writes its data
302        vm2_translator
303            .write_obj(guest_addr, vm2_data)
304            .expect("VM2 failed to write data");
305
306        // Both VMs read back their own data - should be isolated
307        let vm1_read: u64 = vm1_translator
308            .read_obj(guest_addr)
309            .expect("VM1 failed to read data");
310        let vm2_read: u64 = vm2_translator
311            .read_obj(guest_addr)
312            .expect("VM2 failed to read data");
313
314        // Verify isolation: each VM should read its own data
315        assert_eq!(vm1_read, vm1_data, "VM1 should read its own data");
316        assert_eq!(vm2_read, vm2_data, "VM2 should read its own data");
317        assert_ne!(
318            vm1_read, vm2_read,
319            "VM1 and VM2 should have different data (isolation)"
320        );
321
322        // Test buffer operations with different patterns
323        let buffer_addr = GuestPhysAddr::from_usize(0x200);
324        let vm1_buffer = [0xAA; 16]; // Pattern for VM1
325        let vm2_buffer = [0x55; 16]; // Pattern for VM2
326
327        // Both VMs write their patterns
328        vm1_translator
329            .write_buffer(buffer_addr, &vm1_buffer)
330            .expect("VM1 failed to write buffer");
331        vm2_translator
332            .write_buffer(buffer_addr, &vm2_buffer)
333            .expect("VM2 failed to write buffer");
334
335        // Read back and verify isolation
336        let mut vm1_read_buffer = [0u8; 16];
337        let mut vm2_read_buffer = [0u8; 16];
338
339        vm1_translator
340            .read_buffer(buffer_addr, &mut vm1_read_buffer)
341            .expect("VM1 failed to read buffer");
342        vm2_translator
343            .read_buffer(buffer_addr, &mut vm2_read_buffer)
344            .expect("VM2 failed to read buffer");
345
346        assert_eq!(
347            vm1_read_buffer, vm1_buffer,
348            "VM1 should read its own buffer pattern"
349        );
350        assert_eq!(
351            vm2_read_buffer, vm2_buffer,
352            "VM2 should read its own buffer pattern"
353        );
354        assert_ne!(
355            vm1_read_buffer, vm2_read_buffer,
356            "VM buffers should be isolated"
357        );
358
359        // Test that VM1 cannot access VM2's address space (beyond its limit)
360        let vm2_only_addr = GuestPhysAddr::from_usize(crate::test_utils::MEMORY_LEN / 2 + 0x100);
361        let result: AxResult<u32> = vm1_translator.read_obj(vm2_only_addr);
362        assert!(
363            result.is_err(),
364            "VM1 should not be able to access VM2's exclusive address space"
365        );
366    }

370    fn test_cross_page_access() {
371        let translator =
372            MockTranslator::new(PhysAddr::from_usize(0), crate::test_utils::MEMORY_LEN);
373
374        // Test cross-region buffer operations
375        // Place buffer near a region boundary to test multi-region access
376        let cross_region_addr = GuestPhysAddr::from_usize(4096 - 8); // 8 bytes before 4K boundary
377        let test_data = [
378            0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0A, 0x0B, 0x0C, 0x0D, 0x0E,
379            0x0F, 0x10,
380        ]; // 16 bytes
381
382        // Write cross-region data
383        translator
384            .write_buffer(cross_region_addr, &test_data)
385            .expect("Failed to write cross-region buffer");
386
387        // Read cross-region data back
388        let mut read_data = [0u8; 16];
389        translator
390            .read_buffer(cross_region_addr, &mut read_data)
391            .expect("Failed to read cross-region buffer");
392
393        assert_eq!(
394            read_data, test_data,
395            "Cross-region read should match written data"
396        );
397
398        // Test individual byte access across region boundary
399        for (i, &expected_byte) in test_data.iter().enumerate() {
400            let byte_addr = GuestPhysAddr::from_usize(cross_region_addr.as_usize() + i);
401            let read_byte: u8 = translator
402                .read_obj(byte_addr)
403                .expect("Failed to read individual byte");
404            assert_eq!(
405                read_byte, expected_byte,
406                "Byte at offset {} should match",
407                i
408            );
409        }
410    }
