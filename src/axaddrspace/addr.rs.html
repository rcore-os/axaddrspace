26impl page_table_multiarch::riscv::SvVirtAddr for GuestPhysAddr {
27    /// Flushes the TLB for the entire address space. The `_vaddr` parameter is ignored.
28    /// This function always performs a full flush and does not support per-page invalidation.
29    ///
30    /// `nomem` here is safe as `hfence.vvma` does not affect host memory address space.
31    fn flush_tlb(_vaddr: Option<Self>) {
32        unsafe {
33            core::arch::asm!("hfence.vvma", options(nostack, nomem, preserves_flags));
34        }
35    }
